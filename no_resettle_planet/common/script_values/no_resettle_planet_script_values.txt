###########################################################
# 惑星が許容できる最大失業者数
###########################################################
# この計算式で産出される数より惑星内の失業者の数が多い場合は対象の惑星ではPOP成長や製造ができなくなる
# 最大失業者数 = (惑星内の全POP数)/5 ※小数切り捨て
# ただし、5を越える場合は5に固定される
max_unemployed_pop_on_the_planet = {
	base = 0
	add = trigger:num_pops
	divide = 5
	floor = yes
	max = 5
}
