	enable_terra_incognita = yes

	# マウスで選択した国と、その国に対する宗主国、従属国のみをピックアップして色分けする
	###############################################################################################
	###
	# 宗主国、独立国は青
	###
	# マウスで選択した国は宗主国(青)
	color = {
		value = hsv { 0.580 1.0 0.8 1.0 }
		condition = {
			is_same_empire = from
			from = {
				is_overlord = yes
			}
		}
		custom_tooltip = "FEALTY_MODE_OVERLORD1"
	}
	# マウスで選択した国自身は独立国(青)
	color = {
		value = hsv { 0.580 1.0 0.8 1.0 }
		condition = {
			is_same_empire = from
			from = {
				is_overlord = no
				is_subject = no
			}
		}
		custom_tooltip = "FEALTY_MODE_INDEPENDENT"
	}
	# マウスで選択した国の宗主国(青)
	color = {
		value = hsv { 0.580 1.0 0.8 1.0 }
		condition = {
			from = {
				is_subject = yes
				overlord = { is_same_empire = root }
			}
		}
		custom_tooltip = "FEALTY_MODE_OVERLORD2"
	}

	###
	# 従属国
	# ・情報値不足は白
	# ・不忠か秘匿忠誠を誓っているなら赤
	# ・忠実なら緑
	###
	# マウスで選択した国の従属国で情報値不足(黄)
	color = {
		value = rgb { 255 255 0 0 }
		condition = {
			is_subject = yes
			overlord = { is_same_empire = from }
			custom_map_fealty_mode_check_intel_level = { TARGET = from }
		}
		custom_tooltip = "FEALTY_MODE_SUBJECT_UNKNOWN"
	}
	# マウスで選択した国の従属国で秘匿忠誠中(赤)
	color = {
		value = rgb { 255 0 0 0 }
		condition = {
			is_subject = yes
			overlord = { is_same_empire = from }
			custom_map_fealty_mode_check_secret_fealty = yes
		}
		custom_tooltip = "FEALTY_MODE_SUBJECT_SECRET"
	}
	# マウスで選択した国の従属国で不忠(赤)
	color = {
		value = rgb { 255 0 0 0 }
		condition = {
			is_subject = yes
			overlord = { is_same_empire = from }
			is_disloyal_to = overlord
		}
		custom_tooltip = "FEALTY_MODE_SUBJECT_DISLOYAL"
	}
	# マウスで選択した国の従属国で忠実(緑)

		condition = {
			is_subject = yes

		custom_tooltip = "FEALTY_MODE_SUBJECT_LOYAL"
	}

	# マウスで選択した国と同じ宗主国を持つが情報値不足(黄)

		value = rgb { 255 255 0 0 }

			is_subject = yes
